package helloworld;
import java.util.Scanner;

public class Helloworld {

	public static void main(String[] args) {
		Scanner sc =new Scanner(System.in);
		System.out.println("输入姓名和学号和温度:");
		// TODO Auto-generated method stub
		
		String name = sc.next(); //输入字符串
		
		String id = sc.next(); //sc.next可以以空格为分隔
		
		int temp = sc.nextInt(); //输入整数
		
		System.out.printf("你好，%s！ 你的学号是：%s；当前温度值为:%d度。%n",name,id,temp);

		
	}

}

直接在IDE可以运行，也可以生成可执行文件，在文件目录下用命令行。

用命令java -jar hello.jar来执行。
